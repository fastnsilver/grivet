12  import com.fns.grivet.model.Attribute;
13  import com.fns.grivet.model.AttributeType;
14  import com.fns.grivet.model.ClassAttribute;
15  
16  @SpringBootTest
17  public class ClassAttributeRepositoryTest {
18  
19      @Autowired
20      private AttributeRepository attributeRepository;
21  
22      @Autowired
23      private ClassRepository classRepository;
24  
25      @Autowired
26      private ClassAttributeRepository classAttributeRepository;
27  
28      private ClassAttribute[] seedClassAttributes() {
29          Attribute a1 = attributeRepository.save(Attribute.builder().name("birthday").build());
30          Attribute a2 = attributeRepository.save(Attribute.builder().name("socialSecurityNumber").build());
31          com.fns.grivet.model.Class detached = 
32                  com.fns.grivet.model.Class.builder()
33                      .name("PersonDetails")
34                      .description("Sensitive details about an individual").build();
35          com.fns.grivet.model.Class c1 = classRepository.save(detached);
36          ClassAttribute[] expected = { 
37                  ClassAttribute.builder().cid(c1.getId()).aid(a1.getId()).tid(AttributeType.ISO_DATE.getId()).build(),
38                  ClassAttribute.builder().cid(c1.getId()).aid(a2.getId()).tid(AttributeType.VARCHAR.getId()).build()
39                  };
40          expected[0] = classAttributeRepository.save(expected[0]);
41          expected[1] = classAttributeRepository.save(expected[1]);
42          return expected;
43      }
44  
45      @Test
46      public void testFindByCid() {
47          ClassAttribute[] expected = seedClassAttributes();
48          List<ClassAttribute> actual = classAttributeRepository.findByCid(expected[0].getCid());
49          Assertions.assertNotNull(actual, "Expected a list of class attributes returned!");
50          Assertions.assertTrue(actual.size() == 2);
51          Assertions.assertEquals(Arrays.asList(expected), actual);
52      }
53  
54      @Test
55      public void testFindByCidAndAid() {
56          ClassAttribute[] expected = seedClassAttributes();
57          ClassAttribute actual = classAttributeRepository.findByCidAndAid(expected[1].getCid(), expected[1].getAid());
58          Assertions.assertNotNull(actual, "Expected a matching class attribute!");
59          Assertions.assertEquals(expected[1], actual);
60      }
61  
62      @AfterEach
63      public void tearDown() {
64          classAttributeRepository.deleteAll();
65          classRepository.deleteAll();
66          attributeRepository.deleteAll();
67      }

69  }
