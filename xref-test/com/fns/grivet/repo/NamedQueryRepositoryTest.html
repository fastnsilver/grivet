12  @SpringBootTest
13  public class NamedQueryRepositoryTest {
14  
15      @Autowired
16      private NamedQueryRepository repo;
17  
18      @Test
19      // representative; a named query will only execute successfully against known tables/views
20      public void testFindByName() {
21          NamedQuery detached = 
22                  NamedQuery.builder()
23                              .name("totalSpendInLastMonth") 
24                              .type(QueryType.SELECT)
25                              .query("SELECT SUM(AMOUNT) FROM EXPENSE WHERE YEAR(TRANSACTION_DATE) = YEAR(CURRENT_DATE - INTERVAL 1 MONTH) AND MONTH(TRANSACTION_DATE) = MONTH(CURRENT_DATE - INTERVAL 1 MONTH)")
26                              .build();
27          NamedQuery expected = repo.save(detached);
28          NamedQuery actual = repo.findByName("totalSpendInLastMonth");
29          Assertions.assertNotNull(actual, "Expected a matching NamedQuery!");
30          Assertions.assertEquals(expected, actual);
31      }
32  
33      @AfterEach
34      public void tearDown() {
35          repo.deleteAll();
36      }
