62      @Test
63      public void testSaveAndFindEntityById() {
64          Attribute detachedAttribute = Attribute.builder().name("canSpeak").description("Is able to speak?").build();
65          Attribute canSpeak = attributeRepository.save(detachedAttribute);
66          Class c = classRepository.save(Class.builder().name("human").description("A human").build());
67          classAttributeRepository.save(ClassAttribute.builder().aid(canSpeak.getId()).cid(c.getId()).tid(AttributeType.BOOLEAN.getId()).build());
68          LocalDateTime now = LocalDateTime.now();
69          Long eid = entityRepository.newId(c.getId(), now);
70          entityRepository.save(eid, canSpeak, AttributeType.BOOLEAN, true, now);
71  
72          List<EntityAttributeValue> eavList = entityRepository.findByEntityId(eid);
73          Assertions.assertTrue(eavList.size() == 1, "List of entity attribute values should contain only one item!");
74          EntityAttributeValue item = eavList.get(0);
75          Assertions.assertEquals(now.truncatedTo(ChronoUnit.SECONDS), item.getCreatedTime().truncatedTo(ChronoUnit.SECONDS));
76          Assertions.assertEquals(eid, item.getId());
77          Assertions.assertEquals("1", item.getAttributeValue());
78          Assertions.assertEquals(canSpeak.getName(), item.getAttributeName());
79      }

81      @Test
82      public void testFindByCreatedTime() {
83          Attribute detachedAttribute = Attribute.builder().name("maxRpm").description("Maximum rotations per minute.").build();
84          Attribute maxRpm = attributeRepository.save(detachedAttribute);
85          Class c = classRepository.save(Class.builder().name("engine").description("An engine").build());
86          classAttributeRepository.save(ClassAttribute.builder().aid(maxRpm.getId()).cid(c.getId()).tid(AttributeType.INTEGER.getId()).build());
87          LocalDateTime now = LocalDateTime.now();
88          Long eid = entityRepository.newId(c.getId(), now);
89          entityRepository.save(eid, maxRpm, AttributeType.INTEGER, 7500, now);
90  
91          List<EntityAttributeValue> eavList = entityRepository.findByCreatedTime(c.getId(), now.minus(1L, ChronoUnit.SECONDS), now.plus(1L, ChronoUnit.SECONDS));
92          Assertions.assertTrue(eavList.size() == 1, "List of entity attribute values should contain only one item!");
93          EntityAttributeValue item = eavList.get(0);
94          Assertions.assertEquals(now.truncatedTo(ChronoUnit.SECONDS), item.getCreatedTime().truncatedTo(ChronoUnit.SECONDS));
95          Assertions.assertEquals(eid, item.getId());
96          Assertions.assertEquals("7500", item.getAttributeValue());
97          Assertions.assertEquals(maxRpm.getName(), item.getAttributeName());
98      }
