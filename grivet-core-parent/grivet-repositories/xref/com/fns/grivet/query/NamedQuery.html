102     private LocalDateTime updatedTime;

104     @Version
105     @Column
106     private long version;
107 
108     @JsonIgnore

110     @GeneratedValue(strategy = GenerationType.IDENTITY)
111     private Integer id;

113     @JsonProperty("name")

116     private String name;

120     private String query;

122     @JsonProperty("type")

124     private QueryType type;
125 
126     @Valid
127     @ElementCollection(fetch=FetchType.EAGER)
128     @CollectionTable(name="named_query_parameter", joinColumns=@JoinColumn(name="id"))
129     @MapKeyColumn(name="parameter_name")
130     @Column(name="parameter_type", nullable=false)
131     private Map<String, String> params;
132 
133     NamedQuery() {
134         setParams(null);
135     }
136 
137     NamedQuery(String createdBy, String updatedBy, LocalDateTime createdTime, LocalDateTime updatedTime,
138             long version, Integer id, String name, String query, QueryType type, Map<String, String> params) {
139         this.createdBy = createdBy;
140         this.updatedBy = updatedBy;
141         this.createdTime = createdTime;
142         this.updatedTime = updatedTime;
143         this.version = version;
144         this.id = id;
145         this.name = name;
146         setQuery(query);
147         this.type = type;
148         setParams(params);
149     }
150 
151     @JsonProperty("query")
152     public void setQuery(String query) {
153        this.query = query;
154        if (StringUtils.hasText(query)) {
155            if (query.toUpperCase().startsWith("SELECT")) {
156                this.type = QueryType.SELECT;
157            }
158            if (query.toUpperCase().startsWith("CALL")) {
159                this.type = QueryType.SPROC;
160            }
161        } 
162     }
163 
164     @JsonProperty("params")
165     public void setParams(Map<String, String> params) {
166         this.params = params;
167         if (params == null) {
168             this.params = new HashMap<>();
169         }
170     }
171 
172     // only when parameter values are not null, empty, or blank
173     // and parameter definition for named query is not empty
174     // will a MapSqlParameterSource be constructed
175     public MapSqlParameterSource asParameterSource(MultiValueMap<String, ?> parameterValues) {
176         MapSqlParameterSource paramSource = null;
177         if (!CollectionUtils.isEmpty(parameterValues) && !getParams().isEmpty()) {
178             paramSource = new MapSqlParameterSource();
179             Set<Entry<String, String>> paramNameToParamTypeEntries = getParams().entrySet();
180             List<?> values = null; 
181             for (Entry<String, String> entry: paramNameToParamTypeEntries) {
182                 values = parameterValues.get(entry.getKey());
183                 if (!CollectionUtils.isEmpty(values)) {
184                     if (values.size() == 1) {
185                         paramSource.addValue(entry.getKey(), parameterValues.getFirst(entry.getKey()), AttributeType.toSqlType(entry.getValue()));
186                     } else {
187                         paramSource.addValue(entry.getKey(), values);
188                     }
189                 }
190             }
191         }
192         return paramSource;
193     }
194 
195     // only when parameter values are not null, empty, or blank
196     // and parameter definition for named query is not empty
197     // will a List<SqlParameter> be constructed
198     public List<SqlParameter> asSqlParameters(MultiValueMap<String, ?> parameterValues) {
199         List<SqlParameter> sqlParams = null;
200         if (!CollectionUtils.isEmpty(parameterValues) && !getParams().isEmpty()) {
201             sqlParams = new ArrayList<>();
202             Set<Entry<String, String>> paramNameToParamTypeEntries = getParams().entrySet();
203             List<?> values = null; 
204             for (Entry<String, String> entry: paramNameToParamTypeEntries) {
205                 values = parameterValues.get(entry.getKey());
206                 if (!CollectionUtils.isEmpty(values)) {
207                     if (values.size() == 1) {
208                         sqlParams.add(new SqlParameter(entry.getKey(), AttributeType.toSqlType(entry.getValue())));
209                     } else {
210                         throw new IllegalArgumentException("Stored Procedure cannot be executed! Parameter [%s] is not a scalar value!".formatted(entry.getKey()));
211                     }
212                 }
213             }
214         }
215         return sqlParams;
216     }
