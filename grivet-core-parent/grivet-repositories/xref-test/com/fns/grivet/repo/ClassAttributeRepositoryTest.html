13  import com.fns.grivet.model.Attribute;
14  import com.fns.grivet.model.AttributeType;
15  import com.fns.grivet.model.ClassAttribute;
16  
17  @RunWith(SpringRunner.class)
18  @DataJpaTest
19  public class ClassAttributeRepositoryTest {
20      
21      @Autowired
22      private AttributeRepository attributeRepository;
23      
24      @Autowired
25      private ClassRepository classRepository;
26  
27      @Autowired
28      private ClassAttributeRepository classAttributeRepository;
29      
30      private ClassAttribute[] seedClassAttributes() {
31          Attribute a1 = attributeRepository.save(new Attribute("birthday", null));
32          Attribute a2 = attributeRepository.save(new Attribute("socialSecurityNumber", null));
33          com.fns.grivet.model.Class c1 = classRepository.save(new com.fns.grivet.model.Class("PersonDetails", "Sensitive details about an individual", null));
34          ClassAttribute[] expected = { 
35                  new ClassAttribute(c1.getId(), a1.getId(), AttributeType.ISO_DATE.getId(), null),
36                  new ClassAttribute(c1.getId(), a2.getId(), AttributeType.VARCHAR.getId(), null)
37                  };
38          classAttributeRepository.save(expected[0]);
39          classAttributeRepository.save(expected[1]);
40          return expected;
41      }
42      
43      @Test
44      public void testFindByCid() {
45          ClassAttribute[] expected = seedClassAttributes();
46          List<ClassAttribute> actual = classAttributeRepository.findByCid(expected[0].getCid());
47          Assert.assertNotNull("Expected a list of class attributes returned!", actual);
48          Assert.assertTrue(actual.size() == 2);
49          Assert.assertEquals(Arrays.asList(expected), actual);
50      }
51  
52      @Test
53      public void testFindByCidAndAid() {
54          ClassAttribute[] expected = seedClassAttributes();
55          ClassAttribute actual = classAttributeRepository.findByCidAndAid(expected[1].getCid(), expected[1].getAid());
56          Assert.assertNotNull("Expected a matching class attribute!", actual);
57          Assert.assertEquals(expected[1], actual);
58      }

60  }
