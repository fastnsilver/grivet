3   import org.junit.jupiter.api.Assertions;
4   import org.junit.jupiter.api.Test;
5   
6   public class ConstraintTest {
7   
8   	@Test
9   	public void testThatConstraintThrowsExceptionWhenNull() {
10  	    Assertions.assertThrows(IllegalArgumentException.class, () ->  { new Constraint(null); } );
11  	}
12  
13  	@Test
14  	public void testThatConstraintThrowsExceptionWhenEmpty() {
15  	    Assertions.assertThrows(IllegalArgumentException.class, () ->  { new Constraint(new String[] {}); } );
16  	}
17  
18  	@Test
19  	public void testThatLessThan3PartConstraintThrowsException() {
20  	    Assertions.assertThrows(IllegalArgumentException.class, () ->  { new Constraint(new String[] { "foo", "equals" }); } );
21  	}
22  
23  	@Test
24  	public void testThatThreePartConstraintDefinitionSucceeds() {
25  		Constraint c = new Constraint(new String[] { "foo", "equals", "bar" });
26  		Assertions.assertEquals("foo", c.getAttributeName());
27  		Assertions.assertEquals(Operator.EQUALS, c.getOperator());
28  		Assertions.assertArrayEquals(new String[] { "bar" }, c.getValues());
29  		Assertions.assertNull(c.getConjunction());
30  	}
31  
32  	@Test
33  	public void testThatImproperBetweenConstraintDefinitionFails() {
34  	    Assertions.assertThrows(IllegalArgumentException.class, () ->  { new Constraint(new String[] { "foo", "between", "bar" }); } );
35  	}
36  
37  	@Test
38  	public void testThatProperBetweenConstraintDefinitionSucceeds() {
39  		Constraint c = new Constraint(new String[] { "foo", "between", "bar,noogie" });
40  		Assertions.assertEquals("foo", c.getAttributeName());
41  		Assertions.assertEquals(Operator.BETWEEN, c.getOperator());
42  		Assertions.assertArrayEquals(new String[] { "bar", "noogie" }, c.getValues());
43  		Assertions.assertNull(c.getConjunction());
44  	}
45  
46  	@Test
47  	public void testThatProperBetweenConstraintDefinitionWithConjunctionSucceeds() {
48  		Constraint c = new Constraint(new String[] { "foo", "between", "bar,noogie", "and" });
49  		Assertions.assertEquals("foo", c.getAttributeName());
50  		Assertions.assertEquals(Operator.BETWEEN, c.getOperator());
51  		Assertions.assertArrayEquals(new String[] { "bar", "noogie" }, c.getValues());
52  		Assertions.assertEquals(Conjunction.AND, c.getConjunction());
53  	}
54  
55  	@Test
56  	public void testThatConstraintWithProperOrConjunctionSucceeds() {
57  		Constraint c = new Constraint(new String[] { "foo", "startsWith", "f", "or" });
58  		Assertions.assertEquals("foo", c.getAttributeName());
59  		Assertions.assertEquals(Operator.STARTS_WITH, c.getOperator());
60  		Assertions.assertArrayEquals(new String[] { "f" }, c.getValues());
61  		Assertions.assertEquals(Conjunction.OR, c.getConjunction());
62  	}
63  }
