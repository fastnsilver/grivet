16      private static final String BASE_JOIN = "INNER JOIN all_entity_values AS ev ON e.eid = ev.eid ";
17  
18      private final QueryBuilder qb = QueryBuilder.newInstance();
19      private final Map<Integer, Integer> attributeToAttributeTypeMap = new HashMap<>();
20      private final Map<String, Integer> attributeNameToAttributeIdMap = new HashMap<>();
21  
22      @BeforeEach
23      public void setUp() {
24          attributeToAttributeTypeMap.put(1, AttributeType.VARCHAR.getId());
25          attributeToAttributeTypeMap.put(2, AttributeType.INTEGER.getId());
26          attributeNameToAttributeIdMap.put("flavor", 1);
27          attributeNameToAttributeIdMap.put("age", 2);
28      }

30      @Test
31      public void testConstaintWithNullConjunction() {
32          MockHttpServletRequest request = new MockHttpServletRequest();
33          request.setParameter("c", "flavor|equals|Vanilla");
34          DynamicQuery dq = new DynamicQuery(request.getParameterMap().entrySet(), attributeToAttributeTypeMap, attributeNameToAttributeIdMap);
35          Assertions.assertEquals(BASE_JOIN, qb.obtainEntityAttributeValueJoins(dq));
36          Assertions.assertEquals("(ev.aid = ? AND ev.v = ?)", qb.obtainEntityAttributeValueCriteria(dq));
37          Assertions.assertTrue(dq.asSqlParameterValues().length == 2);
38          Assertions.assertEquals(Types.INTEGER, dq.asSqlParameterValues()[0].getSqlType());
39          Assertions.assertEquals(1, dq.asSqlParameterValues()[0].getValue());
40          Assertions.assertEquals(Types.VARCHAR, dq.asSqlParameterValues()[1].getSqlType());
41          Assertions.assertEquals("Vanilla", dq.asSqlParameterValues()[1].getValue());
42      }

44      @Test
45      public void testConstaintWithBetween() {

55          Assertions.assertEquals(Types.INTEGER, dq.asSqlParameterValues()[2].getSqlType());

57          Assertions.assertEquals(46, dq.asSqlParameterValues()[2].getValue());
58      }

80      @Test
81      public void testAndConstraints() {
82          MockHttpServletRequest request = new MockHttpServletRequest();
83          request.setParameter("constraint", "age|equals|35|and");
84          request.setParameter("c", "flavor|startsWith|V");
85          DynamicQuery dq = new DynamicQuery(request.getParameterMap().entrySet(), attributeToAttributeTypeMap, attributeNameToAttributeIdMap);
86          Assertions.assertEquals(BASE_JOIN + "INNER JOIN all_entity_values AS ev1 ON ev.eid = ev1.eid ", qb.obtainEntityAttributeValueJoins(dq));
87          Assertions.assertEquals("(ev.aid = ? AND ev.v = ?) AND (ev1.aid = ? AND ev1.v LIKE ?)", qb.obtainEntityAttributeValueCriteria(dq));
88          Assertions.assertTrue(dq.asSqlParameterValues().length == 4);
89          Assertions.assertEquals(Types.INTEGER, dq.asSqlParameterValues()[0].getSqlType());
90          Assertions.assertEquals(2, dq.asSqlParameterValues()[0].getValue());
91          Assertions.assertEquals(Types.INTEGER, dq.asSqlParameterValues()[1].getSqlType());
92          Assertions.assertEquals(35, dq.asSqlParameterValues()[1].getValue());
93          Assertions.assertEquals(Types.INTEGER, dq.asSqlParameterValues()[2].getSqlType());
94          Assertions.assertEquals(1, dq.asSqlParameterValues()[2].getValue());
95          Assertions.assertEquals(Types.VARCHAR, dq.asSqlParameterValues()[3].getSqlType());
96          Assertions.assertEquals("V%", dq.asSqlParameterValues()[3].getValue());
97      }
