122     private String mutz(String data, Map<String, Object> properties, Set<String> removed) throws JSONException, IOException {
123         String original = payload(data);
124         JSONObject jo = new JSONObject(original);
125         properties.forEach((k,v) -> jo.put(k, v));
126         removed.forEach(k -> jo.remove(k));
127         return jo.toString();
128     }
129 
130     private String payload(String data) throws IOException{
131         Resource r = resolver.getResource("classpath:%s.json".formatted(data));
132         return IOUtils.toString(r.getInputStream(), Charset.defaultCharset());
133     }
