001package com.fns.grivet.model;
002
003import java.util.List;
004
005import org.apache.commons.lang3.builder.ToStringBuilder;
006
007import com.fasterxml.jackson.annotation.JsonCreator;
008import com.fasterxml.jackson.annotation.JsonIgnore;
009import com.fasterxml.jackson.annotation.JsonProperty;
010import com.fasterxml.jackson.annotation.JsonPropertyOrder;
011
012@JsonPropertyOrder(value = { "errors", "method", "uri", "query" })
013public class ErrorResponse {
014    private String method;
015    private String uri;
016    private String query;
017    private List<Object> errors;
018
019    @JsonCreator
020    public ErrorResponse(@JsonProperty String method, @JsonProperty String uri, @JsonProperty String query,
021            @JsonProperty List<Object> errors) {
022        this.method = method;
023        this.uri = uri;
024        this.query = query;
025        this.errors = errors;
026    }
027
028    public String getMethod() {
029        return method;
030    }
031
032    public String getUri() {
033        return uri;
034    }
035
036    public String getQuery() {
037        return query;
038    }
039
040    public List<Object> getErrors() {
041        return errors;
042    }
043
044    @JsonIgnore
045    public String toString() {
046        return ToStringBuilder.reflectionToString(this);
047    }
048
049}
