19  import java.net.URI;

21  import org.json.JSONArray;
22  import org.json.JSONException;

24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.beans.factory.annotation.Value;
26  import org.springframework.cloud.context.config.annotation.RefreshScope;

29  import org.springframework.http.MediaType;
30  import org.springframework.http.ResponseEntity;

35  import org.springframework.web.bind.annotation.PathVariable;
36  import org.springframework.web.bind.annotation.PostMapping;

38  import org.springframework.web.bind.annotation.RequestMapping;
39  import org.springframework.web.bind.annotation.RestController;
40  import org.springframework.web.util.UriComponentsBuilder;

42  import com.fns.grivet.service.ClassRegistryService;

47  /**
48   * Provides end-points for type definition and verification
49   *
50   * @author Chris Phillipson
51   */
52  @Slf4j
53  @RefreshScope
54  @RestController
55  @RequestMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
56  public class ClassRegistryController {
57  
58      @Value("${grivet.register.batch-size:100}")
59      private int batchSize;
60  
61      private final ClassRegistryService classRegistryService;
62  
63      @Autowired
64      public ClassRegistryController(ClassRegistryService classRegistryService) {
65          this.classRegistryService = classRegistryService;
66      }
67  
68      @PreAuthorize("hasAuthority('write:typedef')")
69      @PostMapping("/definition")
70      public ResponseEntity<?> defineType(@RequestBody JSONObject payload) throws IOException {
71          String type = classRegistryService.register(payload);
72          UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
73          log.info("Type [{}] successfully registered!", type);
74          return ResponseEntity.created(ucb.path("/definition/{type}").buildAndExpand(type).toUri()).build();
75      }
76  
77      @PreAuthorize("hasAuthority('write:typedef')")
78      @PostMapping("/definitions")
79      public ResponseEntity<?> defineTypes(@RequestBody JSONArray array) throws IOException, JSONException {
80          int numberOfTypesToRegister = array.length();
81          Assert.isTrue(numberOfTypesToRegister <= batchSize,
82                          "The total number of entries in a type registration request must not exceed %d! Your registration request contained [%d] entries.".formatted(
83                          batchSize, numberOfTypesToRegister));
84          JSONObject payload = null;
85          String type = null;
86          HttpHeaders headers = new HttpHeaders();
87          URI location = null;
88          int errorCount = 0;
89          // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
90          for (int i = 0; i < numberOfTypesToRegister; i++) {
91              try {
92                  payload = array.getJSONObject(i);
93                  type = classRegistryService.register(payload);
94                  location = UriComponentsBuilder.newInstance().path("/definition/{type}").buildAndExpand(type).toUri();
95                  if (numberOfTypesToRegister == 1) {
96                      headers.setLocation(location);
97                  } else {
98                      headers.set("Location[%s]".formatted(String.valueOf(i + 1)), location.toASCIIString());
99                  }
100                 log.info("Type [{}] successfully registered!", type);
101             } catch (Exception e) {
102                 String message = LogUtil.toLog(payload, "Problems registering type! Portion of payload @ index[%d]\n".formatted(i + 1));
103                 log.error(message, e);
104                 if (numberOfTypesToRegister == 1) {
105                     throw e;
106                 }
107                 headers.set("Error[%s]".formatted(String.valueOf(i + 1)), e.getMessage());
108                 errorCount++;
109             }
110         }
111         return new ResponseEntity<>(headers, ((errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED));
112     }
113 
114     @PreAuthorize("hasAuthority('delete:typedef')")
115     @DeleteMapping("/definition/{type}")
116     public ResponseEntity<?> undefineType(
117             @PathVariable("type") String type) {
118         classRegistryService.deregister(type);
119         log.info("Type [{}] successfully deregistered!", type);
120         return ResponseEntity.noContent().build();
121     }
122 
123     @PreAuthorize("hasAuthority('read:typedef')")
124     @GetMapping("/definition/{type}")
125     public ResponseEntity<?> getTypeDefinition(
126             @PathVariable("type") String type) {
127         JSONObject payload = classRegistryService.get(type);
128         String message = LogUtil.toLog(payload, "Successfully retrieved type [%s]\n".formatted(type));
129         log.info(message);
130         return ResponseEntity.ok(payload.toString());
131     }
132 
133     @PreAuthorize("hasAuthority('read:typedef')")
134     @GetMapping("/definitions")
135     public ResponseEntity<?> getAllTypeDefinitions() {
136         return ResponseEntity.ok(classRegistryService.all().toString());
137     }
138 
139 }
