46  /**
47   * Provides end-points for type definition and verification
48   * 
49   * @author Chris Phillipson
50   */
51  @Slf4j
52  @RestController
53  @RequestMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
54  public class ClassRegistryController {
55  
56      @Value("${grivet.register.batch-size:100}")
57      private int batchSize;
58      
59      private final ClassRegistryService classRegistryService;
60      
61      @Autowired
62      public ClassRegistryController(ClassRegistryService classRegistryService) {
63          this.classRegistryService = classRegistryService;
64      }
65      
66      @PreAuthorize("hasAuthority('write:typedef')")
67      @PostMapping("/api/v1/definition")
68      public ResponseEntity<?> defineType(@RequestBody JSONObject payload) throws IOException {
69          String type = classRegistryService.register(payload);
70          UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
71          log.info("Type [{}] successfully registered!", type);
72          return ResponseEntity.created(ucb.path("/api/v1/definition/{type}").buildAndExpand(type).toUri()).build();
73      }
74  
75      @PreAuthorize("hasAuthority('write:typedef')")
76      @PostMapping("/api/v1/definitions")
77      public ResponseEntity<?> defineTypes(@RequestBody JSONArray array) throws IOException, JSONException {
78          int numberOfTypesToRegister = array.length();
79          Assert.isTrue(numberOfTypesToRegister <= batchSize,
80                  String.format(
81                          "The total number of entries in a type registration request must not exceed %d! Your registration request contained [%d] entries.",
82                          batchSize, numberOfTypesToRegister));
83          JSONObject payload = null;
84          String type = null;
85          HttpHeaders headers = new HttpHeaders();
86          URI location = null;
87          int errorCount = 0;
88          // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
89          for (int i = 0; i < numberOfTypesToRegister; i++) {
90              try {
91                  payload = array.getJSONObject(i);
92                  type = classRegistryService.register(payload);
93                  location = UriComponentsBuilder.newInstance().path("/api/v1/definition/{type}").buildAndExpand(type).toUri();
94                  if (numberOfTypesToRegister == 1) {
95                      headers.setLocation(location); 
96                  } else {
97                      headers.set(String.format("Location[%s]",String.valueOf(i+1)), location.toASCIIString());
98                  }
99                  log.info("Type [{}] successfully registered!", type);
100             } catch (Exception e) {
101                 String message = LogUtil.toLog(payload, String.format("Problems registering type! Portion of payload @ index[%d]\n", i+1));
102                 log.error(message, e);
103                 if (numberOfTypesToRegister == 1) {
104                     throw e;
105                 }
106                 headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
107                 errorCount++;
108             }
109         }
110         return new ResponseEntity<>(headers, ((errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED));
111     }
112 
113     @PreAuthorize("hasAuthority('delete:typedef')")
114     @DeleteMapping("/api/v1/definition/{type}")
115     public ResponseEntity<?> undefineType(
116             @PathVariable("type") String type) {
117         classRegistryService.deregister(type);
118         log.info("Type [{}] successfully deregistered!", type);
119         return ResponseEntity.noContent().build();
120     }
121     
122     @PreAuthorize("hasAuthority('read:typedef')")
123     @GetMapping("/api/v1/definition/{type}")
124     public ResponseEntity<?> getTypeDefinition(
125             @PathVariable("type") String type) {
126         JSONObject payload = classRegistryService.get(type);
127         String message = LogUtil.toLog(payload, String.format("Successfully retrieved type [%s]\n", type));
128         log.info(message);
129         return ResponseEntity.ok(payload.toString());
130     }
131     
132     @PreAuthorize("hasAuthority('read:typedef')")
133     @GetMapping("/api/v1/definitions")
134     public ResponseEntity<?> getAllTypeDefinitions() {
135         return ResponseEntity.ok(classRegistryService.all().toString());
136     }
137         
138 }
