55  /**
56   * Provides end-points for type storage and retrieval
57   *
58   * @author Chris Phillipson
59   */
60  @Slf4j
61  @RestController
62  @RequestMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
63  public class EntityController {
64  
65  	@Value("${grivet.store.batch-size:100}")
66  	private int batchSize;
67  
68  	private final EntityService entityService;
69  
70  	private final MetricRegistry metricRegistry;
71  
72  	@Autowired
73  	public EntityController(EntityService entityService, MetricRegistry metricRegistry) {
74  		this.entityService = entityService;
75  		this.metricRegistry = metricRegistry;
76  	}
77  
78  	@Profile("!pipeline")
79  	@PreAuthorize("hasAuthority('write:type')")
80  	@PostMapping("/api/v1/type")
81  	public ResponseEntity<?> createOne(@RequestHeader("Type") String type, @RequestBody JSONObject json) {
82  		Long oid = entityService.create(type, json);
83  		URI location = UriComponentsBuilder.newInstance().path("/api/v1/type").queryParam("oid", oid).build().toUri();
84  		metricRegistry.counter(MetricRegistry.name("store", "create", type, "count")).inc();
85  		log.info("Successfully created type [{}]", type);
86  		return ResponseEntity.created(location).build();
87  	}
88  
89  	@Profile("!pipeline")
90  	@PreAuthorize("hasAuthority('write:type')")
91  	@PostMapping("/api/v1/types")
92  	public ResponseEntity<?> createMultiple(@RequestHeader("Type") String type, @RequestBody JSONArray json) {
93  		int numberOfTypesToCreate = json.length();
94  		Assert.isTrue(numberOfTypesToCreate <= batchSize,
95  				String.format(
96  						"The total number of entries in a request must not exceed %d! Your store request contained [%d] entries.",
97  						batchSize, numberOfTypesToCreate));
98  		int errorCount = 0;
99  		JSONObject jsonObject = null;
100 		HttpHeaders headers = new HttpHeaders();
101 		URI location = null;
102 		Long oid = null;
103 		// allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
104 		for (int i = 0; i < numberOfTypesToCreate; i++) {
105 			try {
106 				jsonObject = json.getJSONObject(i);
107 				oid = entityService.create(type, jsonObject);
108 				location = UriComponentsBuilder.newInstance().path("/api/v1/type").queryParam("oid", oid).build().toUri();
109 				if (numberOfTypesToCreate == 1) {
110 					headers.setLocation(location);
111 				} else {
112 					headers.set(String.format("Location[%s]", String.valueOf(i + 1)), location.toASCIIString());
113 				}
114 				metricRegistry.counter(MetricRegistry.name("store", "create", type, "count")).inc();
115 				log.info("Successfully created type [{}]", type);
116 			} catch (Exception e) {
117 				String message = LogUtil.toLog(jsonObject, String.format("Problems storing type! Portion of payload @ index[%d]\n", i+1));
118 				log.error(message, e);
119 				if (numberOfTypesToCreate == 1) {
120 					throw e;
121 				}
122 				headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
123 				errorCount++;
124 			}
125 		}
126 		return new ResponseEntity<>(headers, ((errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED));
127 	}
128 
129 	@Profile("!pipeline")
130 	@PreAuthorize("hasAuthority('write:type')")
131 	@PatchMapping("/api/v1/type")
132 	public ResponseEntity<?> updateOne(
133 			@RequestParam(value = "oid", required = true) Long oid,
134 			@RequestBody JSONObject json) {
135 		String type = entityService.update(oid, json);
136 		HttpHeaders headers = new HttpHeaders();
137 		URI location = UriComponentsBuilder.newInstance().path("/api/v1/type").queryParam("oid", oid).build().toUri();
138 		headers.setLocation(location);
139 		metricRegistry.counter(MetricRegistry.name("store", "update", type, "count")).inc();
140 		log.info("Successfully updated type [{}]", type);
141 		return ResponseEntity.ok().headers(headers).build();
142 	}
143 
144 	@Profile("!pipeline")
145 	@PreAuthorize("hasAuthority('delete:type')")
146 	@DeleteMapping("/api/v1/type")
147 	public ResponseEntity<?> deleteOne(
148 			@RequestParam(value = "oid", required = true) Long oid) {
149 		String type = entityService.delete(oid);
150 		metricRegistry.counter(MetricRegistry.name("store", "delete", type, "count")).inc();
151 		log.info("Successfully delete type [{}]", type);
152 		return ResponseEntity.ok().build();
153 	}
154 
155 	@PreAuthorize("hasAuthority('read:type')")
156 	@GetMapping("/api/v1/type/{type}")
157 	public ResponseEntity<?> fetch(@PathVariable("type") String type,
158 			@RequestParam(value = "createdTimeStart", required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE_TIME) LocalDateTime createdTimeStart,
159 			@RequestParam(value = "createdTimeEnd", required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE_TIME) LocalDateTime createdTimeEnd,
160 			@RequestParam(value = "noAudit", defaultValue = "false") boolean noAudit,
161 			HttpServletRequest request) throws JsonProcessingException {
162 		LocalDateTime start = createdTimeStart == null ? LocalDateTime.now().minusDays(7) : createdTimeStart;
163 		LocalDateTime end = createdTimeEnd == null ? LocalDateTime.now() : createdTimeEnd;
164 		Assert.isTrue(ChronoUnit.SECONDS.between(start, end) >= 0, "Store request constraint createdTimeStart must be earlier or equal to createdTimeEnd!");
165 		if (noAudit) {
166 		    return ResponseEntity.ok(entityService.findAllByType(type));
167 		} else {
168 		    return ResponseEntity.ok(entityService.findByCreatedTime(type, start, end, request.getParameterMap()));
169 		}
170 	}
171 
172 	@PreAuthorize("hasAuthority('read:type')")
173 	@GetMapping("/api/v1/type")
174 	public ResponseEntity<?> fetchOne(
175 			@RequestParam(value = "oid", required = true) Long oid) {
176 		return ResponseEntity.ok(entityService.findOne(oid));
177 	}

179 }
