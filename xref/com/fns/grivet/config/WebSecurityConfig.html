4   import org.springframework.context.annotation.Configuration;

8   import org.springframework.security.config.annotation.method.configuration.EnableMethodSecurity;
9   import org.springframework.security.config.annotation.web.builders.HttpSecurity;
10  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
11  import org.springframework.security.config.annotation.web.configuration.WebSecurityConfiguration;
12  import org.springframework.security.web.SecurityFilterChain;
13  import org.springframework.security.web.access.expression.DefaultWebSecurityExpressionHandler;
14  
15  import com.okta.spring.boot.oauth.Okta;
16  
17  @Profile("secure")
18  @EnableWebSecurity
19  @Configuration
20  public class WebSecurityConfig {
21  
22      @Bean
23      SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
24  
25          http.authorizeRequests()
26                  // allow anonymous access to the root page
27                  .requestMatchers("/").permitAll()
28                  // all other requests
29                  .anyRequest().authenticated()
30                  .and()
31                  .oauth2ResourceServer().jwt(); // replace .jwt() with .opaqueToken() for Opaque Token case
32  
33          // Send a 401 message to the browser (w/o this, you'll see a blank page)
34          Okta.configureResourceServer401ResponseBody(http);
35          return http.build();
36      }
37  }
