72  	@Transactional
73  	public Long create(String type, JSONObject payload) {
74  		com.fns.grivet.model.Class c = classRepository.findByName(type);
75  		Assert.notNull(c, "Type [%s] is not registered!".formatted(type));
76  		if (c.isValidatable()) {
77  			executeSchemaValidation(type, payload);
78  		}
79  		Set<String> keys = payload.keySet();
80  		Assert.notEmpty(keys, "Type [%s] must declare at least one attribute in create request!".formatted(type));
81  		Object val = null;
82  		Attribute a = null;
83  		ClassAttribute ca = null;
84  		AttributeType at = null;
85  		// entity and entity-attribute-values should be stamped with same date/time
86  		LocalDateTime createdTime = LocalDateTime.now();
87  		Long eid = entityRepository.newId(c.getId(), createdTime);
88  		// TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
89  		for (String key: keys) {
90  			val = payload.get(key);
91  			a = attributeRepository.findByName(key);
92  			Assert.notNull(a, "Attribute [%s] is not registered!".formatted(key));
93  			ca = classAttributeRepository.findByCidAndAid(c.getId(), a.getId());
94  			Assert.notNull(ca, "[%s] is not a valid attribute of [%s]".formatted(key, type));
95  			at = attributeTypeRepository.findById(ca.getTid());
96  			Assert.notNull(at, "Attribute type [%s] is not supported!".formatted(at));
97  			entityRepository.save(eid, a, at, val, createdTime);
98  		}
99  		return eid;
100 	}

102 	@Transactional
103 	public String update(Long eid, JSONObject payload) {
104 		Integer cid = entityRepository.getClassIdForEntityId(eid);
105 		Assert.notNull(cid, "No type registered for entity with oid = [%d]".formatted(eid));
106         com.fns.grivet.model.Class c = classRepository.findById(cid).get();
107 		String type = c.getName();
108 		if (c.isValidatable()) {
109 			executeSchemaValidation(type, payload);
110 		}
111 		// keys (attribute names) of the entity we wish to update; there should
112 		// be at least one to update!
113 		Set<String> detachedKeys = payload.keySet();
114 		Assert.notEmpty(detachedKeys,
115                 "Type [%s] must declare at least one attribute in update request!".formatted(type));
116 
117 		// get currently persisted entity's attribute values; make sure the oid
118 		// passed actually exists!
119 		String currentEntity = findById(eid);
120 
121 		// keys (attribute names) of the currently persisted entity
122 		JSONObject persistentObject = new JSONObject(currentEntity);
123 		Set<String> persistentKeys = persistentObject.keySet();
124 
125 		Object val = null;
126 		Attribute a = null;
127 		ClassAttribute ca = null;
128 		AttributeType at = null;
129 		// entity and entity-attribute-values should be stamped with same date/time
130 		LocalDateTime createdTime = LocalDateTime.now();
131 
132 		// reconcile detached entity's (payload's) attribute-values
133 		// for each persistentKey
134 		// * check that payload has a value for that key
135 		// * if value exists; persist it as an update
136 		// * if value does not exist; use current entity's value and re-persist
137 		for (String key : persistentKeys) {
138 			val = payload.get(key);
139 			if (val == null) {
140 				val = persistentObject.get(key);
141 			}
142 			a = attributeRepository.findByName(key);
143 			Assert.notNull(a, "Attribute [%s] is not registered!".formatted(key));
144 			ca = classAttributeRepository.findByCidAndAid(c.getId(), a.getId());
145 			Assert.notNull(ca, "[%s] is not a valid attribute of [%s]".formatted(key, type));
146 			at = attributeTypeRepository.findById(ca.getTid());
147 			Assert.notNull(at, "Attribute type [%s] is not supported!".formatted(at));
148 			entityRepository.save(eid, a, at, val, createdTime);
149 		}
150 		return type;
151 	}

154 	public String delete(Long eid) {
155 		Integer cid = entityRepository.getClassIdForEntityId(eid);
156 		Assert.notNull(cid, "No type registered for entity with oid = [%d]".formatted(eid));
157         com.fns.grivet.model.Class c = classRepository.findById(cid).get();

159 		entityRepository.delete(eid);
160 		return type;
161 	}

163 	@Transactional(readOnly=true)
164 	public String findByCreatedTime(String type, LocalDateTime createdTimeStart, LocalDateTime createdTimeEnd,
165 			Map<String, String[]> parameters) throws JsonProcessingException {
166 		com.fns.grivet.model.Class c = classRepository.findByName(type);
167 		Assert.notNull(c, "Type [%s] is not registered!".formatted(type));
168 		Map<Integer, Integer> attributeToAttributeTypeMap = generateAttributeToAttributeTypeMap(c);
169 		List<Attribute> attributes = Lists.newArrayList(attributeRepository.findAll());
170 		Map<String, Integer> attributeNameToAttributeIdMap = attributes.stream().collect(Collectors.toMap(Attribute::getName, Attribute::getId));
171 		Set<Entry<String, String[]>> params = parameters == null ? null : parameters.entrySet();
172 		DynamicQuery query = new DynamicQuery(params, attributeToAttributeTypeMap, attributeNameToAttributeIdMap);
173 		List<EntityAttributeValue> rows = null;
174 		if (query.hasConstraints()) {
175 			rows = entityRepository.executeDynamicQuery(c.getId(), query);
176 		} else {
177 			rows = entityRepository.findByCreatedTime(c.getId(), createdTimeStart, createdTimeEnd);
178 		}
179 		return mapRows(attributeToAttributeTypeMap, rows).toString();
180 	}

182 	@Transactional(readOnly=true)
183 	public String findById(Long eid) {
184 		List<EntityAttributeValue> rows = entityRepository.findByEntityId(eid);
185 		if (rows == null) {
186 			throw new ResourceNotFoundException("No entity exists with oid =[%d]".formatted(eid));
187 		}
188 		Integer cid = entityRepository.getClassIdForEntityId(eid);
189         com.fns.grivet.model.Class c = classRepository.findById(cid).get();
190 		Map<Integer, Integer> attributeToAttributeTypeMap = generateAttributeToAttributeTypeMap(c);
191 		return mapRows(attributeToAttributeTypeMap, rows).getJSONObject(0).toString();
192 	}

194 	@Transactional(readOnly = true)
195 	public String findAllByType(String type) {
196 		com.fns.grivet.model.Class c = classRepository.findByName(type);
197 		Assert.notNull(c, "Type [%s] is not registered!");
198 		List<EntityAttributeValue> rows = entityRepository.findAllEntitiesByCid(c.getId());
199 		Map<Integer, Integer> attributeToAttributeTypeMap = generateAttributeToAttributeTypeMap(c);
200 		return mapRows(attributeToAttributeTypeMap, rows).toString();
201 	}
202 
203 	protected JSONArray mapRows(Map<Integer, Integer> attributeToAttributeTypeMap, List<EntityAttributeValue> rows) {
204 		JSONArray jsonArray = new JSONArray();
205 		String current = null;
206 		String previous = null;
207 		JSONObject jsonObject = null;
208 		for (EntityAttributeValue row: rows) {
209 			current = String.format("%s>%d", row.getCreatedTime().toString(), row.getId());
210 			if (!current.equals(previous)) {
211 				jsonObject = new JSONObject();
212 				jsonArray.put(jsonObject);
213 			}
214 			Integer tid = attributeToAttributeTypeMap.get(row.getAttributeId());
215 			jsonObject.put(row.getAttributeName(), ValueHelper.getValue(attributeTypeRepository.findById(tid), row.getAttributeValue()));
216 			previous = current;
217 		}
218 		return jsonArray;
219 	}
220 
221 	private void executeSchemaValidation(String type, JSONObject payload) {
222 		ProcessingReport report = schemaValidator.validate(type, payload);
223 		if (!report.isSuccess()) {
224 			List<Object> errors = new ArrayList<>();
225 			report.forEach(pm -> errors.add(pm.asJson()));
226 			throw new SchemaValidationException(errors);
227 		}
228 	}
229 
230 	private Map<Integer, Integer> generateAttributeToAttributeTypeMap(com.fns.grivet.model.Class c) {
231 		List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
232 		return cas.stream().collect(Collectors.toMap(ClassAttribute::getAid, ClassAttribute::getTid));
233 	}
