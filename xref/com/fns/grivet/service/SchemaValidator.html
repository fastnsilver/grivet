16  package com.fns.grivet.service;
17  
18  import com.fasterxml.jackson.databind.JsonNode;
19  import com.fasterxml.jackson.databind.ObjectMapper;
20  import com.fns.grivet.repo.ClassRepository;
21  import com.github.fge.jsonschema.core.exceptions.ProcessingException;
22  import com.github.fge.jsonschema.core.report.ProcessingReport;
23  import com.github.fge.jsonschema.main.JsonSchema;
24  import com.github.fge.jsonschema.main.JsonSchemaFactory;
25  
26  import org.json.JSONObject;
27  import org.springframework.beans.factory.annotation.Autowired;
28  import org.springframework.stereotype.Component;
29  import org.springframework.util.Assert;
30  
31  import java.io.IOException;
32  
33  @Component
34  class SchemaValidator {
35      
36      private final ClassRepository classRepository;
37      private final ObjectMapper objectMapper;
38      
39      @Autowired
40      public SchemaValidator(ClassRepository classRepository, ObjectMapper objectMapper) {
41          this.classRepository = classRepository;
42          this.objectMapper = objectMapper;
43      }
44  
45      public ProcessingReport validate(String type, JSONObject payload) throws SchemaValidationException {
46          try {
47              JsonSchemaFactory factory = JsonSchemaFactory.byDefault();
48              Assert.notNull(type, String.format("Type [%s] does not exist! Schema cannot be retrieved!", type));
49              com.fns.grivet.model.Class c = classRepository.findByName(type);
50              Assert.notNull(c, String.format("Persistent model for type [%s], does not exist!", type));
51              JsonNode schemaAsJsonNode = objectMapper.readTree(c.getJsonSchema());
52              Assert.notNull(schemaAsJsonNode, "Corrupt persistent Schema representation!");
53              final JsonSchema schema = factory.getJsonSchema(schemaAsJsonNode);
54              JsonNode instance = objectMapper.readTree(payload.toString());
55              Assert.notNull(schema, String.format("Schema not found for type [%s]!", type));
56              Assert.notNull(instance, String.format("Problem generating JsonNode for payload\n\n%s!", payload.toString()));
57              return schema.validate(instance);
58          } catch (ProcessingException | IOException e) {
59              throw new SchemaValidationException(String.format("Problem validating [%s] against JSON Schema", type), e);
60          }
61      }
62      
63  }
