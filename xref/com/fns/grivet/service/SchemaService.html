16  package com.fns.grivet.service;
17  
18  import java.time.LocalDateTime;
19  
20  import org.apache.commons.lang3.StringUtils;
21  import org.json.JSONObject;
22  import org.springframework.beans.factory.annotation.Autowired;
23  import org.springframework.stereotype.Service;
24  import org.springframework.transaction.annotation.Transactional;
25  import org.springframework.util.Assert;
26  
27  import com.fns.grivet.model.User;
28  import com.fns.grivet.repo.ClassRepository;
29  
30  @Service
31  public class SchemaService {
32  
33  	private static final String ID = "id";
34  	private static final String SCHEMA = "$schema";
35  	private static final String TYPE = "type";
36  	private static final String PROPERTIES = "properties";
37  
38  	private final ClassRepository classRepository;
39  
40  	@Autowired(required=false)
41  	private SecurityFacade securityFacade;
42  
43  	@Autowired
44  	public SchemaService(ClassRepository classRepository) {
45  		this.classRepository = classRepository;
46  	}
47  
48  	public boolean isJsonSchema(JSONObject payload) {
49  		boolean result = false;
50  		String schema = payload.optString(SCHEMA);
51  		String type = payload.optString(TYPE);
52  		String id = payload.optString(ID);
53  		String props = payload.optString(PROPERTIES);
54  		if ("http://json-schema.org/draft-04/schema#".equals(schema)
55  				&& "object".equals(type)
56  				&& StringUtils.isNotBlank(id)
57  				&& StringUtils.isNotBlank(props)) {
58  			result = true;
59  		}
60  		return result;
61  	}
62  
63  	@Transactional
64  	public com.fns.grivet.model.Class linkSchema(JSONObject payload) {
65  		String type = payload.getString(ID);
66  		Assert.notNull(type, "JSON Schema must declare an id!");
67  		com.fns.grivet.model.Class c = classRepository.findByName(type);
68  		Assert.notNull(c, String.format("Type [%s] must be registered before linking a JSON Schema!", type));
69  		c.setValidatable(true);
70  		c.setJsonSchema(payload.toString());
71  		c.setUpdatedTime(LocalDateTime.now());
72  		classRepository.save(c);
73  		return c;
74  	}
75  
76  	@Transactional
77  	public com.fns.grivet.model.Class unlinkSchema(String type) {
78  		com.fns.grivet.model.Class c = classRepository.findByName(type);
79  		Assert.notNull(c, String.format("Type [%s] must be registered before unlinking a JSON Schema!", type));
80  		c.setValidatable(false);
81  		c.setJsonSchema(null);
82  		c.setUpdatedTime(LocalDateTime.now());
83  		User user = securityFacade != null ? securityFacade.getCurrentUser(): null;
84  		c.setUpdater(user);
85  		classRepository.save(c);
86  		return c;
87  	}
88  
89  }
