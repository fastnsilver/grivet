16  package com.fns.grivet.service;
17  
18  import java.time.LocalDateTime;
19  import java.util.List;
20  import java.util.Set;
21  import java.util.stream.StreamSupport;
22  
23  import org.json.JSONArray;
24  import org.json.JSONObject;
25  import org.springframework.beans.factory.annotation.Autowired;
26  import org.springframework.dao.DuplicateKeyException;
27  import org.springframework.stereotype.Service;
28  import org.springframework.transaction.annotation.Transactional;
29  import org.springframework.util.Assert;
30  
31  import com.fns.grivet.model.Attribute;
32  import com.fns.grivet.model.AttributeType;
33  import com.fns.grivet.model.ClassAttribute;
34  import com.fns.grivet.repo.AttributeRepository;
35  import com.fns.grivet.repo.AttributeTypeRepository;
36  import com.fns.grivet.repo.ClassAttributeRepository;
37  import com.fns.grivet.repo.ClassRepository;
38  
39  @Service
40  public class ClassRegistryService {
41  
42  	private static final String TYPE = "type";
43  	private static final String DESCRIPTION = "description";
44  	private static final String ATTRIBUTES = "attributes";
45  
46  	private final AttributeRepository attributeRepository;
47  	private final AttributeTypeRepository attributeTypeRepository;
48  	private final ClassRepository classRepository;
49  	private final ClassAttributeRepository classAttributeRepository;
50  
51  	@Autowired
52  	public ClassRegistryService(AttributeRepository attributeRepository,
53  			AttributeTypeRepository attributeTypeRepository, ClassRepository classRepository,
54  			ClassAttributeRepository classAttributeRepository) {
55  		this.attributeRepository = attributeRepository;
56  		this.attributeTypeRepository = attributeTypeRepository;
57  		this.classRepository = classRepository;
58  		this.classAttributeRepository = classAttributeRepository;
59  	}
60  
61  	@Transactional
62  	// USE WITH CAUTION! Consider that w/ FK, type and all related data will be destroyed
63  	public void deregister(String type) {
64  		com.fns.grivet.model.Class c = classRepository.findByName(type);
65  		if (c != null) {
66  			classRepository.delete(c);
67  		}
68  	}
69  
70  	@Transactional
71  	public String register(JSONObject payload) {
72  		String type = payload.getString(TYPE);
73  		String description = payload.optString(DESCRIPTION);
74  		com.fns.grivet.model.Class persistentClass = classRepository.findByName(type);
75  		if (persistentClass == null) {
76  			LocalDateTime createdTime = LocalDateTime.now();
77  			com.fns.grivet.model.Class detachedClass = com.fns.grivet.model.Class.builder().name(type).description(description).build();
78  			detachedClass.setCreatedTime(createdTime);
79  			persistentClass = classRepository.save(detachedClass);
80  			JSONObject attributes = payload.getJSONObject(ATTRIBUTES);
81  			Set<String> attributeNames = attributes.keySet();
82  			Assert.notEmpty(attributeNames, "[%s] must declare at least one attribute in registration request!".formatted(type));
83  			Attribute persistentAttribute = null;
84  			String at = null;
85  			AttributeType attributeType = null;
86  			// TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
87  			for (String attributeName: attributeNames) {
88  				persistentAttribute = attributeRepository.findByName(attributeName);
89  				if (persistentAttribute == null) {
90  					Attribute detachedAttribute = Attribute.builder().name(attributeName).build();
91  					detachedAttribute.setCreatedTime(createdTime);
92  					persistentAttribute = attributeRepository.save(detachedAttribute);
93  				}
94  				at = attributes.getString(attributeName);
95  				Assert.notNull(at, "[%s].[%s] must declare an attribute type in registration request!".formatted(type, attributeName));
96  				attributeType = attributeTypeRepository.findByType(at);
97  				Assert.notNull(attributeType, "Attribute type [%s] is not supported!".formatted(at));
98  				ClassAttribute detachedClassAttribute = 
99  				        ClassAttribute.builder()
100 				            .cid(persistentClass.getId())
101 				            .aid(persistentAttribute.getId())
102 				            .tid(attributeType.getId()).build();
103 				detachedClassAttribute.setCreatedTime(createdTime);
104 				classAttributeRepository.save(detachedClassAttribute);
105 			}
106 		} else {
107 			throw new DuplicateKeyException("Type [%s] already registered!".formatted(type));
108 		}
109 		return type;
110 	}
111 
112 	@Transactional(readOnly=true)
113 	public JSONObject get(String type) {
114 		com.fns.grivet.model.Class c = classRepository.findByName(type);
115 		Assert.notNull(c, "Type [%s] is not registered!".formatted(type));
116 		JSONObject result = new JSONObject();
117 		result.put("type", type);
118 		String description = c.getDescription();
119 		result.put("description", description != null ? description: "");
120 		List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
121 		Assert.notEmpty(cas, "Type [%s] does not have any attributes registered!".formatted(type));
122 		JSONObject attributes = new JSONObject();
123 		result.put("attributes", attributes);
124 		Attribute a = null;
125 		AttributeType at = null;
126 		for (ClassAttribute ca: cas) {
127             a = attributeRepository.findById(ca.getAid()).get();
128 			Assert.notNull(a, String.format("Attribute id [%s] is not registered!", ca.getAid()));
129 			at = attributeTypeRepository.findById(ca.getTid());
130 			Assert.notNull(a, String.format("Attribute Type id [%s] is not registered!", ca.getTid()));
131 			attributes.put(a.getName(), at.getType());
132 		}
133 		return result;
134 	}
135 
136 	@Transactional(readOnly=true)
137 	public JSONArray all() {
138 		JSONArray result = new JSONArray();
139 		Iterable<com.fns.grivet.model.Class> iterable = classRepository.findAll();
140 		Assert.notNull(iterable, "No types are registered!");
141 		StreamSupport
142 		    .stream(iterable.spliterator(), false)
143 		        .forEach(c -> result.put(get(c.getName())));
144 		return result;
145 	}
146 
147 }
